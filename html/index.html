못본 페이지에 checkbox radio
있을 경우 자동으로 for id적용		 // <label><input type="checkbox"></label> 수정되지 않은 이런형식이 남아있을 경우 스크립트로 해결
document.addEventListener("DOMContentLoaded", function () {
const usedIds = new Set();
const inputs = document.querySelectorAll('input[type="checkbox"], input[type="radio"]');

inputs.forEach((input, index) => {
const label = input.closest('label') || document.querySelector(`label[for="${input.id}"]`);

// input에 id가 없으면 고유한 id 생성 후 추가
if (!input.id) {
let uniqueId;
do {
uniqueId = `${input.type}_${Math.floor(Math.random() * 1000000)}_${index}`;
} while (usedIds.has(uniqueId));

input.id = uniqueId;
usedIds.add(uniqueId);
} else {
usedIds.add(input.id);
}

// label이 존재하고 label에 for가 없는 경우, input의 id를 for로 추가
if (label && !label.hasAttribute('for')) {
label.setAttribute('for', input.id);
}
});
});
